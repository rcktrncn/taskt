taskt command line reference

nothing (Only script file path. Do not specify any other parameters)
	Run (Execute) script file.
	ex.) taskt "C:\scripts\myscript.xml"

-o
	Open script file.
	ex.) taskt -o "C:\scripts\myscript.xml"

-oh
	Open script file as unsaved file.
	ex.) taskt -oh "C:\scripts\myscript.xml"

-r
	Run (Execute) script file.
	ex.) taskt -r "C:\scripts\myscript.xml"
	
-e
	Run (Execute) script file.
	ex.) taskt -e "C:\scripts\myscript.xml"

-s
	Specify settings file path.
	ex.) taskt -s "C:\taskt\mysettings.xml"
